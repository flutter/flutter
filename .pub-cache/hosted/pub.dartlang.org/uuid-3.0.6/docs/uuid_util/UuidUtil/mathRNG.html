static Uint8List mathRNG({int seed = -1}) {
  final b = Uint8List(16);
  final rand = (seed == -1) ? _random : Random(seed);

  for (var i = 0; i < 16; i++) {
    b[i] = rand.nextInt(256);
  }

  return b;
}
